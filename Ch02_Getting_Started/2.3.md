### 2.3-2

```
MERGE(A, p, q, r)
n1 = q - p + 1
n2 = r - q
let L[n1] and R[n2]
for i = 1 to n1
    L[i] = A[p + i - 1]
for j = 1 to n2
    L[j] = A[q + i]

for k = 1 to n1 and l = 1 to n2, i = 0
    if L[k] > R[l]
        A[p + i] = R[l++]
    else
        A[p + i] = L[k++]
    i++

if k == n1
    for l to n2
        A[p + i] = R[l++]
else
    for k to n1
        A[p + i] = L[k++]

```
